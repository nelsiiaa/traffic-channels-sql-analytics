The Search paradox. Search has a higher bounce rate than Social — yet converts at 5.5× the rate. Users arrive with purchase intent. They bounce if irrelevant, but buy when the product matches. Social drives clicks without commitment.

Direct traffic, assumed to signal brand loyalty, actually shows the lowest return rate (1.32 visits). Email drives the longest sessions — avg 280s vs 59s for Social.

SQL — selected queries
01_table_base.sql fixed: ORDER BY in window fn
-- Event-level → session-level transformation -- FIX: removed ARRAY_AGG OVER (ORDER BY) — not supported in BigQuery -- pageview_count via COUNT(DISTINCT CASE WHEN) in GROUP BY instead CREATE OR REPLACE VIEW `web-analytics-494818.web_analitycs.table_base` AS WITH raw_events AS ( SELECT user_pseudo_id, COALESCE( (SELECT ep.value.int_value FROM UNNEST(event_params) ep WHERE ep.key = 'ga_session_id'), 0 ) AS session_id, event_name, event_timestamp, event_date, COALESCE((SELECT ep.value.string_value FROM UNNEST(event_params) ep WHERE ep.key = 'source'), '(direct)') AS traffic_source, COALESCE((SELECT ep.value.string_value FROM UNNEST(event_params) ep WHERE ep.key = 'medium'), '(none)') AS traffic_medium, COALESCE((SELECT ep.value.int_value FROM UNNEST(event_params) ep WHERE ep.key = 'engagement_time_msec'), 0) / 1000.0 AS engagement_time_sec, device.category AS device_category FROM `bigquery-public-data.ga4_obfuscated_sample_ecommerce.events_*` WHERE _TABLE_SUFFIX BETWEEN '20201101' AND '20210131' ) SELECT user_pseudo_id, session_id, traffic_source, traffic_medium, device_category, event_date, COUNT(DISTINCT CASE WHEN event_name = 'page_view' THEN event_timestamp END) AS pageview_count, SUM(engagement_time_sec) AS engagement_time_sec, MAX(CASE WHEN event_name = 'purchase' THEN 1 ELSE 0 END) AS purchase_done FROM raw_events WHERE session_id != 0 GROUP BY user_pseudo_id, session_id, traffic_source, traffic_medium, device_category, event_date
02_traffic_channels.sql unified channel grouping
-- Bounce rate, conversion rate, revenue by source category WITH channel_grouped AS ( SELECT *, CASE WHEN traffic_source IN ('google','bing','yahoo') AND traffic_medium = 'cpc' THEN 'Search' WHEN traffic_medium = 'organic' THEN 'Organic' WHEN traffic_source LIKE '%facebook%' OR traffic_source LIKE '%youtube%' THEN 'Social media' WHEN traffic_medium = 'email' THEN 'Email' WHEN traffic_source = '(direct)' AND traffic_medium = '(none)' THEN 'Direct' ELSE 'Other' END AS source_category FROM `web-analytics-494818.web_analitycs.table_base` ) SELECT source_category, COUNT(DISTINCT session_id) AS total_sessions, ROUND(100.0 * SUM(CASE WHEN pageview_count = 1 AND engagement_time_sec < 10 THEN 1 ELSE 0 END) / COUNT(*), 2) AS bounce_rate_pct, ROUND(100.0 * SUM(purchase_done) / COUNT(*), 2) AS conversion_rate_pct FROM channel_grouped GROUP BY source_category ORDER BY total_sessions DESC
03_channel_device_breakdown.sql new — per feedback
-- Device × channel segmentation (missing from original submission) SELECT CASE WHEN traffic_source IN ('google','bing') AND traffic_medium = 'cpc' THEN 'Search' WHEN traffic_source LIKE '%facebook%' OR traffic_source LIKE '%youtube%' THEN 'Social' WHEN traffic_medium = 'email' THEN 'Email' WHEN traffic_source = '(direct)' THEN 'Direct' ELSE 'Other' END AS source_category, device_category, COUNT(DISTINCT session_id) AS sessions, ROUND(AVG(engagement_time_sec), 1) AS avg_engagement_sec, ROUND(100.0 * SUM(purchase_done) / COUNT(*), 2) AS conversion_pct FROM `web-analytics-494818.web_analitycs.table_base` GROUP BY source_category, device_category ORDER BY source_category, sessions DESC
